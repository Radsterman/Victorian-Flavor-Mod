ai_strategy_conservative_agenda = {
	icon = "gfx/interface/icons/ai_strategy_icons/conservative_agenda.dds"
	
	type = political
	
	revolution_aversion = {
		value = 75
	}
	
	min_law_chance_to_pass = {
		value = 30

		if = {
			limit = { 				
				OR = {
					any_political_movement = {
						desired_law ?= scope:law.type
						political_movement_radicalism >= 75
					}
					AND = {
						has_journal_entry = je_government_petition
						var:desired_law_var	?= scope:law			
					}
				}
			}
			
			multiply = 0.33
		}			
	}
	
	max_progressiveness = {
		value = 50
		if = {
			limit = { any_interest_group = { is_insurrectionary = yes is_powerful = yes } }
			multiply = 1.5
		}
	}
	
	max_regressiveness = {
		value = 25
		if = {
			limit = { any_interest_group = { is_insurrectionary = yes is_powerful = yes } }
			multiply = 1.5
		}
	}
	
	change_law_chance = { 
		value = 2.5
		
		if = {
			limit = { any_interest_group = { is_insurrectionary = yes } }
			add = 7.5
		}
		
		if = {
			limit = { any_interest_group = { is_insurrectionary = yes is_powerful = yes } }
			add = 7.5
		}
		
		if = {
			limit = { 
			    any_political_movement = {
					political_movement_radicalism >= 70
				}
			}
			add = 7.5
		}

		if = {
			limit = { has_journal_entry = je_government_petition  }
			add = 25
		}			
	}		
	
	pro_interest_groups = {
		ig_rural_folk
		ig_devout
		ig_landowners
		ig_armed_forces
		ig_petty_bourgeoisie
	}
	
	institution_scores = {
		institution_colonial_affairs = {	
			value = 150			
		}				
		institution_social_security = {	
			value = 50			
		}
		institution_workplace_safety = {	
			value = 50			
		}		
		institution_schools = {	
			value = 100			
		}		
		institution_police = {	
			value = 150			
		}		
		institution_health_system = {	
			value = 100		
		}		
		institution_home_affairs = {	
			value = 150		
		}				
	}

	possible = {
		always = yes
	}
	
	weight = {
		value = 20	
		
		if = {
			limit = {
				exists = ruler
				ruler = {
					OR = {
						has_ideology = ideology:ideology_traditionalist
						has_monarchist_ideology = yes
					}
				}
			}
			
			add = 10
		}	

		if = {
			limit = { 
				exists = c:FRA
				c:FRA = this
				ruler = {
					has_variable = orleanist_leader
				}
			}
			add = 10
		}
		
		if = {
			limit = {
				has_law = law_type:law_council_republic	
			}
			multiply = 0.25
		}	
		
	}
}
