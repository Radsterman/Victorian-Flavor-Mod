je_establish_los_altos_button = {
	name = establish_los_altos_button
	desc = establish_los_altos_button_desc

	visible = {
		this = c:UCA
		NOT = { has_variable = los_altos_var }
		NOT = { exists = c:GUA }
	}

	possible = {
		has_law = law_type:law_federal
		OR = {
			custom_tooltip = {
				text = establish_los_altos_requirements_tt
				NOT = { has_variable = gua_civil_peace_var }
			}
			has_law = law_type:law_multicultural
		}
	}

	effect = {
		custom_tooltip = {
			text = establish_los_altos_tt
			set_variable = los_altos_var
		}
		scope:guatemala_state = {
			add_loyalists_in_state = {
				value = 0.15
			}
		}
		set_variable = { name = restore_order_action_var days = 35 }
	}

	ai_chance = {
		base = 100
	}
}

je_nicaragua_intervention_central_america_button = {
	name = nicaragua_intervention_central_america_button
	desc = nicaragua_intervention_central_america_button_desc

	visible = { 
		this = c:UCA
		NOT = { has_variable = je_central_america_setup_nic_var }
		exists = c:NIC
		c:NIC ?= {
			is_subject_of = ROOT
		}
	}

	possible = {
		custom_tooltip = {
			text = uca_restore_order_action_tt
			NOT = { has_variable = restore_order_action_var }
		}
		custom_tooltip = {
			text = nicaragua_subject_frca_tt
			c:NIC ?= {
				is_subject_of = ROOT
			}
		}
		custom_tooltip = {
			text = invade_nicaragua_frca_stab_tt
			this = c:UCA
			exists = c:NIC
			has_variable = gua_support_var
			has_variable = hon_support_var
			has_variable = els_support_var
		}
		custom_tooltip = {
			text = invade_nicaragua_frca_cos_support_tt
			this = c:UCA
			exists = c:NIC
			has_variable = cos_support_var
		}
	}

	effect = {
		custom_tooltip = {
			text = invade_nicaragua_frca_tt
			create_diplomatic_play = {
				name = nicaraguan_intervention
				type = dp_annex_war
				escalation = 20
				target_state = s:STATE_NICARAGUA.region_state:NIC
			
			}
		}
		set_variable = { name = restore_order_action_var days = 35 }
	}

	ai_chance = {
		base = 0
	}
}
