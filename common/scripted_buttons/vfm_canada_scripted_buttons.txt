je_unify_two_canadas_gbr_button = {
	name = "je_unify_two_canadas_button"
	desc = "je_unify_two_canadas_button_desc"

	visible = { 
		exists = c:ONT
		exists = c:QUE
		c:ONT ?= {
			is_subject_of = root
			has_revolution = no
		}
		c:QUE ?= {
			is_subject_of = root
			has_revolution = no
		}
	}

	possible = {
		c:ONT ?= {
			is_subject_of = root
			has_technology_researched = nationalism
			has_revolution = no
			is_ai = yes
		}
		c:QUE ?= {
			is_subject_of = root
			has_technology_researched = nationalism
			has_revolution = no
			is_ai = yes
		}
	}

	effect = {
		c:ONT = { 
			annex = c:QUE
			add_primary_culture = cu:franco_canadian
			#set_variable = canadas_unified
			if = {
				limit = {
					NOT = { exists = c:CAN }
				}
				change_tag = CAN
			}
		}
	}
	
	ai_chance = {
		base = 5
	}
}

				annex = c:COL
			}

		#trigger event selling rupert's land

		base = 5

}

je_found_british_columbia_button = {
	name = "je_found_british_columbia_button"
	desc = "je_found_british_columbia_button_desc"

	visible = {
		OR = {
			AND = {
				exists = c:COL
				c:COL = {
					is_subject_of = root
					NOT = { is_subject_type = subject_type_dependency }
				}
			}
			has_state_in_state_region = STATE_BRITISH_COLOMBIA
		}
	}

	possible = {
		OR = {
			AND = {
				exists = c:COL
				c:COL = {
					is_subject_of = root
				}
			}
			has_state_in_state_region = STATE_BRITISH_COLOMBIA
		}
	}

	effect = {
		if = {
			limit = {
				AND = {
					exists = c:COL
					c:COL = {
						is_subject_of = root
					}
				}
			}
			create_diplomatic_pact = {
				country = c:COL
				type = dependency
			}
		}
		if = {
			limit = {
				has_state_in_state_region = STATE_BRITISH_COLOMBIA
				NOT = {
					exists = c:COL
					c:COL = {
						is_subject_of = root
					}
				}
			}
			create_country = {
				tag = COL
				origin = root
				state = s:STATE_BRITISH_COLOMBIA.region_state:root
			}
			create_diplomatic_pact = {
				country = c:COL
				type = dependency
			}
		}
	}
	
	ai_chance = {
		base = 5
	}
}
