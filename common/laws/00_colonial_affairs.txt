law_no_colonial_affairs = {
	group = lawgroup_colonization
	
	icon = "gfx/interface/icons/law_icons/no_colonial_affairs.dds"
	
	progressiveness = 0	
	
	on_enact = {
		if = {
			limit = {
				any_scope_state = {
					has_modifier = frontier_colonialism_modifier
				}
			}
			custom_tooltip = {
				text = frontier_remove_tt
				every_scope_state = {
					limit = {
						has_modifier = frontier_colonialism_modifier
					}
					remove_modifier = frontier_colonialism_modifier
				}
			}
		}
	}
	
	pop_support = {
		value = 0
	}

	# AI will never fully dismantle institutions
	ai_will_do = {
		always = no
	}	

    ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}

law_frontier_colonialism = {
	group = lawgroup_colonization
	
	icon = "gfx/interface/icons/law_icons/colonial_resettlement.dds"
	
	progressiveness = 5
	
	on_enact = {
		if = {
			limit = {
				any_scope_state = {
					NOT = {
						any_neighbouring_state = {
							OR = {
								is_incorporated = yes
								any_neighbouring_state = {
									is_incorporated = yes
									owner = scope:colony_founder
								}
							}
							owner = scope:colony_founder
						}
						has_modifier = frontier_colonialism_modifier
					}
				}
			}
			custom_tooltip = {
				text = frontier_add_tt
				every_scope_state = {
					limit = {
						NOT = {
							any_neighbouring_state = {
								OR = {
									is_incorporated = yes
									any_neighbouring_state = {
										is_incorporated = yes
										owner = ROOT
									}
								}
								owner = ROOT
							}
							has_modifier = frontier_colonialism_modifier
						}
					}
					add_modifier = frontier_colonialism_modifier
				}
			}
		}
	}

	unlocking_technologies = {
		colonization
	}
	
	modifier = {
		state_migration_pull_unincorporated_mult = 0.25
	}
	
	institution = institution_colonial_affairs 

	pop_support = {
		value = 0
		# Officers like colonialism
		add = {
			desc = "POP_OFFICERS"			
			if = {
				limit = {
					owner = { 
						has_law = law_type:law_no_colonial_affairs 
					} 
					is_pop_type = officers
				}
				value = 0.25
			}
		}
	}
	
	ai_will_do = {
		AND = {
			#AI doesn't change to frontier
			#OR = {
			#	has_law = law_type:law_colonial_resettlement
			#	has_law = law_type:law_colonial_exploitation
			#	has_law = law_type:law_frontier_colonialism
			#}
			any_neighbouring_state = {
				owner = {
					is_country_type = decentralized
				}
			}
		}
	}

    ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}
