		country_suppression_ig_attraction_mult = -0.5
		country_tech_spread_mult = -0.15
	}

	}

	unlocking_technologies = {
		political_agitation
	}

	pop_support = {
		value = 0
	}

	can_impose = { 
		is_subject_of = scope:initiator	
	}

		value = 0

			}
			add = 750
		}
	}
}

	}

		value = 0
		add = {
			desc = "POP_ACADEMICS"			
			if = {
				limit = {

						NOT = { 
							has_law = law_type:law_protected_speech
