je_acw_countdown = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	group = je_group_historical_content

	immediate = {
		set_variable = { name = free_states value = num_free_states }
		set_variable = { name = slave_states value = num_slave_states }
		set_variable = { name = free_territories value = num_free_territories }
		set_variable = { name = slave_territories value = num_slave_territories }
		set_variable = { name = proportion_free  value = prop_free_slave }
		
        set_variable = { name = acw_countdown_next value = 0 }
		
		set_variable = { name = acw_countdown_divisiveness value = 0 }
		set_variable = { name = divisiveness_multiplier value = acw_countdown_divisiveness_multiplier }
	}
	
	on_weekly_pulse = {
		effect = {
			set_variable = { name = free_states value = num_free_states }
			set_variable = { name = slave_states value = num_slave_states }
			set_variable = { name = free_territories value = num_free_territories }
			set_variable = { name = slave_territories value = num_slave_territories }
			set_variable = { name = proportion_free  value = prop_free_slave }

			calculate_acw_countdown_progress = yes
			
			set_variable = { name = divisiveness_multiplier value = acw_countdown_divisiveness_multiplier }
		}
	}

	on_monthly_pulse = {
		effect = {
			change_variable = { name = acw_progress_var add = var:acw_countdown_next }
		}
		random_events = {
			# American Civil War buildup
			200 = 0
			10 = acw_events.1 # The Abolitionist's Martyrdom
			10 = acw_events.3 # The Wilmot Proviso
		}
	}

	on_yearly_pulse = {
		random_events = {
			40 = 0
			20 = acw_events.2 # The Fugitive Slave
			20 = acw_events.5 # Bleeding Kansas
			20 = acw_events.6 # The Caning of Charles Sumner
		}
	}

	complete = {
		calc_true_if = {
			amount >= 3
			custom_tooltip = {
				text = "je_acw_countdown_complete_2"
				NOT = {
					any_in_global_list = {
						variable = usa_slave_plantations_list
						NAND = {
							has_active_production_method = pm_steam_rail_transport
							custom_tooltip = {
								text = "je_acw_countdown_complete_2_2"
								OR = {
									has_active_production_method = automatic_irrigation_building_coffee_plantation
									has_active_production_method = automatic_irrigation_building_cotton_plantation
									has_active_production_method = automatic_irrigation_building_dye_plantation
									has_active_production_method = automatic_irrigation_building_opium_plantation
									has_active_production_method = automatic_irrigation_building_sugar_plantation
									has_active_production_method = automatic_irrigation_building_tea_plantation
									has_active_production_method = automatic_irrigation_building_tobacco_plantation
									has_active_production_method = automatic_irrigation_building_banana_plantation
									has_active_production_method = automatic_irrigation_building_silk_plantation
									has_active_production_method = automatic_irrigation_building_vineyard_plantation
								}
							}
						}
					}
				}
			}
			custom_tooltip = {
				text = "je_acw_countdown_complete_3"
				OR = {
					calc_true_if = {
						amount >= 3
						market.mg:tobacco.market_goods_imports > market.mg:tobacco.market_goods_exports
						market.mg:sugar.market_goods_imports > market.mg:sugar.market_goods_exports
						market.mg:fabric.market_goods_imports > market.mg:fabric.market_goods_exports
					}
					calc_true_if = {
						amount >= 3
						market.mg:tobacco.market_goods_imports > market.mg:tobacco.market_goods_production
						market.mg:sugar.market_goods_imports > market.mg:sugar.market_goods_production
						market.mg:fabric.market_goods_imports > market.mg:fabric.market_goods_production
					}
				}
			}
			custom_tooltip = {
				text = "je_acw_countdown_complete_4"
				NOT = {
					any_in_global_list = {
						variable = great_powers_list
						any_trade_route = {
							NOT = { actor_market.owner = root }
							exporter = root.market
							NAND = {
								goods = g:tobacco
								goods = g:sugar
								goods = g:fabric
							}
						}
					}
				}
			}
			ig:ig_landowners = {
				is_marginal = yes
			}
			#custom_tooltip = {
			#	text = "je_acw_countdown_complete_1"
			#	always = no
			#}
		}
	}
	
	on_complete = {
		custom_tooltip = {
			text = "je_acw_countdown_complete_general"
		}
		set_variable = je_acw_countdown_complete
		#remove_modifier = missouri_compromise
		trigger_event = { id = slavery_law_events.7 days = 0 popup = yes }
	}

	fail = {
		OR = {
			custom_tooltip = {
				text = je_acw_countdown_fail_1
				var:acw_progress_var <= 0
			}
			custom_tooltip = {
				text = je_acw_countdown_fail_2
				var:acw_progress_var >= 200
			}
			custom_tooltip = {
				text = je_acw_countdown_fail_3
				AND = {
					OR = {
						has_revolution_over_law_type = law_type:law_slavery_banned
						has_revolution_over_law_type = law_type:law_slave_trade
						has_revolution_over_law_type = law_type:law_legacy_slavery
						has_revolution_over_law_type = law_type:law_debt_slavery
					}		
				
					OR = {
						# CSA
						custom_tooltip = {
							text = acw_landowner_revolt
							any_country = {
								is_revolutionary = yes
								country_definition = root.country_definition
								ig:ig_landowners = {
									is_in_government = yes
								}
							}
						}
						# FSA
						custom_tooltip = {
							text = acw_antislaver_revolt
							ig:ig_landowners = {
								is_in_government = yes
							}
							NOT = { has_law = law_type:law_slavery_banned }
							NOT = { is_enacting_law = law_type:law_slavery_banned }
							any_country = {
								is_revolutionary = yes
								country_definition = root.country_definition
							}
						}
					}
				}
			}
		}
	}

	on_fail = {
		if = {
			limit = {
				any_country = {
					is_revolutionary = yes
					country_definition = root.country_definition
					ig:ig_landowners = {
						is_in_government = yes
					}				
				}
			}
			set_variable = csa_will_spawn_var
		}
		custom_tooltip = {
			text = acw_will_happen_tt
			trigger_event = { id = acw_je_events.1 }
		}
	}

	current_value = {
		value = root.var:acw_progress_var
	}

	goal_add_value = {
		value = 200
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}

	group = je_group_historical_content

	complete = {
		NOT = {
			any_scope_state = {
				has_modifier = unreconstructed_state
				is_incorporated = no
			}
		}
		NOT = {
			any_scope_state = {
				state_region = {
					is_homeland = cu:dixie
				}
				loyalist_fraction = {
					value < 0.2
					culture = cu:dixie
				}
			}
		}
	}

	on_complete = {
		custom_tooltip = {
			text = dont_lose_dixie_culture
			set_variable = acw_keep_dixie
		}
		custom_tooltip = {
			text = reconstruction_progress
			change_variable = {
				name = reconstruction_var
				add = 1
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			180 = 0
			20 = acw_followup_events.6
			20 = acw_followup_events.7
			20 = acw_followup_events.8
		}
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = {
					is_homeland = cu:dixie
				}
			}
		}
	}

	timeout = 4380
}

je_acw_wild_wild_west = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	group = je_group_historical_content

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		NOT = {
			any_scope_state = {
				state_region = {
					OR = {
						is_homeland = cu:dixie
						is_homeland = cu:yankee
					}
				}
				devastation > 0.1
				turmoil > 0.1
			}
		}
	}

	on_complete = {
		custom_tooltip = {
			text = acw_no_outlaw_crisis
		}
		custom_tooltip = {
			text = reconstruction_progress
			change_variable = {
				name = reconstruction_var
				add = 1
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = crime_events.1 # train robbery
			10 = wild_west.1 # bank robberies
			10 = wild_west.2 # Raids on the Iron Horse
			10 = wild_west.3 # Gunfight at the Corral
		}
		effect = {
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							state_region = {
								OR = {
									is_homeland = cu:dixie
									is_homeland = cu:yankee
								}
							}
							devastation > 0.1
							turmoil > 0.1
						}
					}
				}
				change_variable = {
					name = je_acw_wild_west_var
					add = 1
				}
			}
		}
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = {
					OR = {
						is_homeland = cu:dixie
						is_homeland = cu:yankee
					}
				}
			}
		}
	}

	current_value = {
		value = var:je_acw_wild_west_var
	}
	
	goal_add_value = {
		add = 60
	}

	progressbar = yes

	timeout = 4380
}
