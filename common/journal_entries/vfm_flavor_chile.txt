	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_crises

	modifiers_while_active = {
		fra_republican_unrest
	}

	immediate = {
		c:CHL = {
			random_scope_character = {
				limit = {
					has_variable = is_bilbao
				}
				save_scope_as = bilbao_scope
		    }
		}
	}	

	complete = {
		AND = {
	        custom_tooltip = {
			    text = voting_franchise_yes_tt
			    country_has_voting_franchise = yes
			    }
			NOT = { has_law = law_type:law_single_party_state }   
		    }
	        AND = { 
	            OR = {   
	            ig:ig_intelligentsia = {
			        is_powerful = yes
			        is_in_government = yes
			            }                
	            ig:ig_trade_unions = {
			        is_powerful = yes
			        is_in_government = yes
			        }
			    }
	        } 
	    }               

	on_complete = { 
	    trigger_event = { 
	        id = journal_flavor_chile.12  
	    }
    }

	on_fail = { 
	    trigger_event = { 
	        id = journal_flavor_chile.11  
	    }
    }  

    timeout = 1095

	weight = 10000

	should_be_pinned_by_default = yes  
}
