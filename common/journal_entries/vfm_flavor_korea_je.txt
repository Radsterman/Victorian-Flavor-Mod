je_sadae = {
	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"
	group = je_group_internal_affairs
	immediate = {
		c:CHI = {
			save_scope_as = chi_scope
		}
	}
	invalid = {
		has_global_variable = china_shatters
		AND = {
			NOT = {
				c:CHI = {
					has_diplomatic_pact = {
						who = ROOT
						type = tributary
						is_initiator = yes
					}
				}
			}
			NOT = { is_subject_of = c:CHI }
		}	
	}
	complete = {
		NOT = { 
			top_overlord = {
				OR = {
					is_at_war = yes
					is_diplomatic_play_committed_participant = yes
				}
			} 
		}
		OR = {
			top_overlord = {
				custom_tooltip = {
					text = kor_chi_relations_tt
					relations:root >= relations_threshold:friendly
				}
				custom_tooltip = {
					text = kor_chi_obligation_tt
					owes_obligation_to = root
				}
			}
			custom_tooltip = {
				text = kor_chi_gdp_tt
				GDP >= gdp_chi
			}
		}
	}

	fail = {
		top_overlord = {
			custom_tooltip = {
				text = kor_chi_relations_bad_tt
				relations:root < relations_threshold:neutral
			}
			is_at_war = no
		}
	}

	on_complete = {
		trigger_event = {
			id = kor.3
		}
	}

	on_fail = {
		c:CHI = {
			trigger_event = {
				id = kor.4
			}
		}
	}
	should_be_pinned_by_default = yes
}

			}
		}
	}
