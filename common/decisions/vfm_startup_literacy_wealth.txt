# Use this file to set literacy and wealth pop changes post-calculation since it can't be changed in the history file

stupify = {

	is_shown = {
		THIS = c:PPU
		exists = c:PPU
		NOT = {
			has_global_variable = stupify_complete
		}
	}

	possible = {
		THIS = c:PPU
		exists = c:PPU
		NOT = {
			has_global_variable = stupify_complete
		}
	}

	when_taken = {
		# Slaves are mostly illiterate
		every_country = {
			every_scope_pop = {
				limit = {
					is_pop_type = slaves
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						divide = 7.5
					}
				}
			}
		}
		#Decentralized peoples without written Christianity/Islam
		every_country = {
			limit = {
				is_country_type = decentralized
			}
			every_scope_pop = {
				limit = {
					NOT = {
						OR = {
							has_pop_religion = sunni
							has_pop_religion = shiite
							has_pop_religion = anglican
							has_pop_religion = lutheran
							has_pop_religion = reformed
						}
					}
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 0.1
					}
				}
			}
		}
		# Laborers and Peasants illiterate
		every_country = {
			every_scope_pop = {
				limit = {
					OR = {
						pop_employment_building_group = bg_subsistence_agriculture
						pop_employment_building_group = bg_subsistence_ranching
					}
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 0.85
					}
				}
			}
		}
		# Jews slight literacy buff
		every_country = {
			every_scope_pop = {
				limit = {
					has_pop_religion = jewish
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.2
					}
				}
			}
		}
		# Catholics slight literacy nerf
		every_country = {
			every_scope_pop = {
				limit = {
					has_pop_religion = catholic
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 0.9
					}
				}
			}
		}
		# Discriminated pops tend to be more illiterate in literate countries
		every_country = {
			every_scope_pop = {
				limit = {
					pop_is_discriminated = yes
					NOT = { has_pop_religion = jewish }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 0.67
					}
				}
			}
		}
		#Each level of education institution increases literacy of primary pops in incorporated states
		every_country = {
			limit = {
				institution_investment_level = { institution = institution_schools value >= 1 }
			}
			every_scope_pop = {
				limit = {
					pop_is_discriminated = no
					state = { is_incorporated = yes }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.05
					}
				}
			}
		}
		every_country = {
			limit = {
				institution_investment_level = { institution = institution_schools value >= 2 }
			}
			every_scope_pop = {
				limit = {
					pop_is_discriminated = no
					state = { is_incorporated = yes }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.05
					}
				}
			}
		}
		every_country = {
			limit = {
				institution_investment_level = { institution = institution_schools value >= 3 }
			}
			every_scope_pop = {
				limit = {
					pop_is_discriminated = no
					state = { is_incorporated = yes }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.05
					}
				}
			}
		}
		#Free Prussia buff since there's no other way to give this without giving absurd universities, institutions, etc.
		every_country = {
			limit = {
				THIS = c:PRU
			}
			every_scope_pop = {
				limit = {
					pop_is_discriminated = no
					state = { is_incorporated = yes }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.55
					}
				}
			}
		}
		#Buffs for German minors
		every_country = {
			limit = {
				country_has_primary_culture = cu:north_german
				country_has_state_religion = rel:lutheran
				NOT = { THIS = c:PRU }
			}
			every_scope_pop = {
				limit = {
					pop_is_discriminated = no
					state = { is_incorporated = yes }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.5
					}
				}
			}
		}
		#Buffs for Baltic states
		every_country = {
			limit = {
				OR = {
					THIS = c:CRL
					THIS = c:LVN
					THIS = c:EST
				}
			}
			every_scope_pop = {
				limit = {
					state = { is_incorporated = yes }
					#literacy_rate <= .21
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 5
					}
				}
			}
		}
		#Regional France Literacy
		every_state = {
			limit = {
				OR = {
					state_region = s:STATE_CHAMPAGNE
					state_region = s:STATE_LORRAINE
					state_region = s:STATE_ALSACE_LORRAINE
					state_region = s:STATE_FRANCHE_COMTE
					state_region = s:STATE_ILE_DE_FRANCE
				}
			}
			every_scope_pop = {
				limit = {
					pop_is_discriminated = no
					state = { is_incorporated = yes }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.4
					}
				}
			}
		}
		every_state = {
			limit = {
				OR = {
					state_region = s:STATE_PICARDY
					state_region = s:STATE_CALAIS
					state_region = s:STATE_NORMANDY
					state_region = s:STATE_RHONE
				}
			}
			every_scope_pop = {
				limit = {
					pop_is_discriminated = no
					state = { is_incorporated = yes }
				}
				set_pop_literacy = {
					literacy_rate = {
						value = literacy_rate
						multiply = 1.25
					}
				}
			}
		}
		set_global_variable = stupify_complete
	}

	ai_chance = {
		base = 100
	}
}
