puq_colonization = {

	is_shown = {
	    exists = c:CHL
		this = c:CHL    
		NOT = { has_variable = puq_colonized }
        }
        
	possible = {
		is_at_war = no
         institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 2
		}
	}

	when_taken = {
	    hidden_effect = { set_variable = puq_colonized }	
        trigger_event = { 
			id = journal_flavor_chile.1
		}			

	ai_chance = {
		base = 100
		}
	}
}
		
rapa_nui = {

	is_shown = {
	    exists = c:CHL
		this = c:CHL    
		exists = c:RPN
		c:RPN = {
            is_subject = no
            }		
		NOT = { has_variable = rapa_nui_colonized }
        }

	possible = {
		institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 3
		}
	    has_interest_marker_in_region = region_oceania	
	    has_technology_researched = civilizing_mission	
	    c:RPN = {
            is_subject = no
        }
	}

	when_taken = {
	    hidden_effect = { set_variable = rapa_nui_colonized }	
        trigger_event = { 
			id = journal_flavor_chile.2
		}		

	ai_chance = {
		base = 100
		}
	}
}
