ohiggins1 = {

	is_shown = {
	    exists = c:CHL
		this = c:CHL  
		year < 1841  
		NOT = { has_variable = ohiggins_returned }
        }
        
	possible = {
		is_at_war = no
		year < 1841
        custom_tooltip = {
			text = is_portales_desc
        ig:ig_petty_bourgeoisie.leader = { NOT = { has_variable = is_portales } } 
            }
        ig:ig_intelligentsia = {
			is_powerful = no
			is_in_government = no
			}               
	    }

	when_taken = {
	    custom_tooltip = return_ohiggins
		hidden_effect = { set_global_variable = ohiggins_returned }		
		c:CHL = {
		    create_character = {
			    first_name = "bernardo"
			    last_name = "ohiggins"
			    ig_leader = yes
			    historical = yes
			    culture = cu:south_andean
			    ruler = no
			    birth_date = 1778.08.20
			    hq = region_la_plata
			    is_general = yes			    
			    dna = dna_bernardo_ohiggins
			    interest_group = ig_armed_forces
			    ideology = ideology_republican_leader
			    traits = {
				    persistent
				    brave
				    celebrity_commander
				    basic_offensive_planner
				    inspirational_orator
				    sickly
				    }
			        on_created = {
				        set_variable = {
					        name = is_ohiggins
				        }
			        }				    
			    }
		    }
        trigger_event = { 
			id = journal_flavor_chile.4
		    }			

	ai_chance = {
		base = 70
		}
	}
}
