abolish_pompous_court_decision = {

	is_shown = {
		this = c:PER
		has_modifier = per_pompous_court
    }
        
	possible = {
		has_modifier = per_pompous_court
		has_technology_researched = egalitarianism
	}

	when_taken = {
		
		remove_modifier = per_pompous_court
		
    }			

	ai_chance = {
		base = 100
	}
	
}

abolish_khozeimeh_amirdom_decision = {

	is_shown = {
		has_state_in_state_region = STATE_KHORASAN 
		any_state = {
			has_modifier = per_khozeimeh_amirdom
		}
    }
        
	possible = {
		owns_entire_state_region = STATE_KHORASAN 
		NOT = { has_law = law_type:law_local_police }
		NOT = { has_law = law_type:law_devolved }
		has_technology_researched = central_archives
	}

	when_taken = {
		
		every_state = {
			limit = {
				has_modifier = per_khozeimeh_amirdom
			}
			remove_modifier = per_khozeimeh_amirdom
		}
		
    }			

	ai_chance = {
		base = 100
	}
}

form_khorasan_decision = {

	is_shown = {
		has_state_in_state_region = STATE_KHORASAN 
		OR = {
			this = c:ZFL
			this = c:SDL
		}
		NOT = { has_variable = become_khorasan_var }
    }
        
	possible = {
		owns_entire_state_region = STATE_KHORASAN 
		has_technology_researched = nationalism
	}

	when_taken = {
		
		custom_tooltip = {
			text = become_khorasan_tt
			set_variable = become_khorasan_var
		}
		
    }			

	ai_chance = {
		base = 100
	}
}
