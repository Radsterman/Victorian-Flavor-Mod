namespace = alaska

# An Offer for Alaska
alaska.1 = {
	type = country_event
	placement = ROOT

	title = alaska.1.t
	desc = alaska.1.d
	flavor = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:alaska_seller = {
						c:RUS ?= this
					}
				}
				desc = alaska.1.f_ru
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = alaska.1.f
			}
		}
	}

	duration = 3

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	trigger = {
	}

	immediate = {
		# Scopes set in alaska.4
	}

	option = {
		name = alaska.1.a
		default_option = yes
		add_modifier = {
			name = alaska_rus_big_money
			months = 12
			is_decaying = no
		}
		scope:alaska_buyer = {
			trigger_event = { id = alaska.2 days = 0 popup = yes }
		}
		show_as_tooltip = {
			random_state = {
				limit = {
					state_region = s:STATE_ALASKA
					owner = scope:alaska_seller
				}
				set_state_owner = scope:alaska_buyer
			}
			change_relations = {
				country = scope:alaska_buyer
				value = 10
			}
		}
		ai_chance = {
			base = 75
		}
	}
	option = {
		name = alaska.1.b
		scope:alaska_buyer = {
			trigger_event = { id = alaska.3 days = 0 popup = yes }
		}
		show_as_tooltip = {
			change_relations = {
				country = scope:alaska_buyer
				value = -20
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					scope:alaska_seller = { 
						NOT = {
							has_variable = alaska_sale_cooldown #Unsolicited offer
						}
					}
				}
				add = 25
			}
		}
	}
}

alaska.2 = {
	type = country_event
	placement = ROOT

	title = alaska.2.t
	desc = alaska.2.d
	flavor = alaska.2.f

	duration = 3

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
	}

	immediate = {
		# Scopes set in alaska.4
	}

	option = {
		name = alaska.2.a
		default_option = yes
		add_modifier = {
			name = alaska_usa_big_money
			months = 12
		}
		scope:alaska_seller = {
			set_variable = successful_sale_of_alaska
		}
		random_state = {
			limit = {
				state_region = s:STATE_ALASKA
				OR = {
				    owner = scope:alaska_seller
					owner.top_overlord = scope:alaska_seller
				}
			}
			set_state_owner = scope:alaska_buyer
		}
		change_relations = {
			country = scope:alaska_seller
			value = 10
		}
	}
}

# Making an Offer
alaska.4 = {
	type = country_event
	placement = ROOT

	title = alaska.4.t
	desc = alaska.4.d
	flavor = alaska.4.f

	duration = 3

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
	}

	immediate = {
		root = {
			save_scope_as = alaska_buyer
		}
		random_state = {
			limit = {
				state_region = s:STATE_ALASKA
				owner = {
					NOT = {
						is_country_type = decentralized
					}
				}
			}
			save_scope_as = alaska_scope
			owner = {
				top_overlord = {
					save_scope_as = alaska_seller
			    }
			}
		}
	}

	option = {
		name = alaska.4.a
		ai_chance = {
			base = 1
		}
		custom_tooltip = alaska.4.tt1 # 3 million over a year
		scope:alaska_seller = {
			trigger_event = {
				id = alaska.1
				days = 0
				popup = yes
			}
		}
	}
	option = { # perhaps another time
		name = alaska.4.c
		default_option = yes
		ai_chance = {
			base = 0
		}
		scope:alaska_seller = {
			post_notification = alaska_negotiations_fail
		}
	}
}
