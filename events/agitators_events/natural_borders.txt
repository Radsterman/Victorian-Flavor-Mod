namespace = natural_borders

# start event
natural_borders.1 = {
	type = country_event
	placement = root
	title = natural_borders.1.t
	desc = natural_borders.1.d
	flavor = natural_borders.1.f

	duration = 3

	dlc = dlc004

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
		# triggered by je_french_natural_borders
	}

	immediate = {
		random_interest_group = {
			limit = {
				is_powerful = yes
				is_in_government = yes
				OR = {
					is_interest_group_type = ig_armed_forces
					leader = {
						OR = {
							has_ideology = ideology:ideology_jingoist_leader
							has_ideology = ideology:ideology_ethno_nationalist
							has_ideology = ideology:ideology_fascist
							has_ideology = ideology:ideology_authoritarian
							has_ideology = ideology:ideology_bonapartist
						}
					}
				}
			}
			leader = { save_scope_as = borders_leader }
		}
		s:STATE_SAVOY = {
			random_scope_state = {
				save_scope_as = savoy_state
			}
		}
		s:STATE_NORTH_RHINE = {
			random_scope_state = {
				save_scope_as = rhineland_state
			}
		}
		s:STATE_FLANDERS = {
			random_scope_state = {
				save_scope_as = flanders_state
			}
		}
		c:FRA = { save_scope_as = france_scope }
	}

	option = {
		name = natural_borders.1.a
		default_option = yes
		ai_chance = {
			base = 1
		}
		if = {
			limit = {
				NOT = { owns_entire_state_region = STATE_LUXEMBOURG }
			}
			s:STATE_LUXEMBOURG = {
				add_claim = ROOT
			}
		}
		if = {
			limit = {
				NOT = { owns_entire_state_region = STATE_WALLONIA }
			}
			s:STATE_WALLONIA = {
				add_claim = ROOT
			}
		}
		if = {
			limit = {
				NOT = { owns_entire_state_region = STATE_FLANDERS }
			}
			s:STATE_FLANDERS = {
				add_claim = ROOT
			}
		}
		if = {
			limit = {
				NOT = { owns_entire_state_region = STATE_RHINELAND }
			}
			s:STATE_RHINELAND = {
				add_claim = ROOT
			}
		}
		if = {
			limit = {
				NOT = { owns_entire_state_region = STATE_NORTH_RHINE }
			}
			s:STATE_NORTH_RHINE = {
				add_claim = ROOT
			}
		}
		if = {
			limit = {
				NOT = { owns_entire_state_region = STATE_SAVOY }
			}
			s:STATE_SAVOY = {
				add_claim = ROOT
			}
		}
		add_modifier = {
			name = france_natural_borders_pursuit
			months = very_long_modifier_time
			# country_infamy_decay_mult
		}
		every_country = {
			limit = {
				NOT = { THIS = ROOT }
				any_scope_state = {
					OR = {
						state_region = s:STATE_WALLONIA
						state_region = s:STATE_FLANDERS
						state_region = s:STATE_LUXEMBOURG
						state_region = s:STATE_RHINELAND
						state_region = s:STATE_NORTH_RHINE
						state_region = s:STATE_SAVOY
					}
				}
			}
			trigger_event = { id = natural_borders.4 }
		}
		hidden_effect = {
			every_country = {
				limit = {
					has_diplomatic_relevance = c:FRA
					NOT = { THIS = ROOT }
				}
				post_notification = fra_natural_borders_notification
			}
		}
	}
	option = { # imperialist IG's unhappy
		name = natural_borders.1.b
		ai_chance = {
			base = 0
		}
		custom_tooltip = {
			text = dismiss_natural_borders_tt
			set_variable = reject_natural_borders
		}
		every_interest_group = {
			limit = {
				is_powerful = yes
				is_in_government = yes
				OR = {
					is_interest_group_type = ig_armed_forces
					leader = {
						OR = {
							has_ideology = ideology:ideology_jingoist_leader
							has_ideology = ideology:ideology_ethno_nationalist
							has_ideology = ideology:ideology_fascist
							has_ideology = ideology:ideology_authoritarian
							has_ideology = ideology:ideology_bonapartist
						}
					}
				}
			}
			add_modifier = {
				name = no_natural_borders
				months = long_modifier_time
				is_decaying = yes
				# interest_group_approval_add
			}
		}
	}
}

# France annexes Savoy
natural_borders.3 = {
	type = country_event
	placement = scope:savoy_state
	title = natural_borders.3.t
	desc = {
		first_valid = {
			triggered_desc = {
				desc = natural_borders.3.d
				trigger = {
					exists = scope:nice_state
				}
			}
			triggered_desc = {
				desc = natural_borders.3.d2
				trigger = {
					always = yes
				}
			}
		}
	}
	flavor = natural_borders.3.f

	duration = 3

	dlc = dlc004

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
		# triggered by france_annex_savoy
	}

	immediate = {
		random_country = {
			limit = {
				is_at_war = no
				any_scope_state = {
					state_region = s:STATE_SAVOY
					is_split_state = no
					is_capital = no
				}
			}
			save_scope_as = savoy_country
			ruler = {
				save_scope_as = savoy_ruler
			}
			random_scope_state = {
				limit = {
					state_region = s:STATE_SAVOY
					is_split_state = no
					is_capital = no
				}
				save_scope_as = savoy_state
			}
			if = {
				limit = {
					any_scope_state = {
						state_region = s:STATE_PROVENCE
					}
				}
				random_scope_state = {
					limit = {
						state_region = s:STATE_PROVENCE
					}
					save_scope_as = nice_state
				}
			}
		}
	}

	option = { # incorporate Savoy into France
		name = natural_borders.3.a
		default_option = yes
		if = {
			limit = {
				is_owed_obligation_by = scope:savoy_country
			}
			scope:savoy_country = {
				set_owes_obligation_to = {
					country = ROOT
					setting = no
				}
			}
		}
		if = {
			limit = {
				has_global_variable = fra_savoy_sardinia
			}
			show_as_tooltip = {
				set_owes_obligation_to = {
					country = scope:savoy_country
					setting = yes
				}
			}
		}
		scope:savoy_state = {
			set_state_owner = ROOT
			set_state_type = incorporated
		}
		if = {
			limit = {
				exists = scope:nice_state
			}
			scope:nice_state = {
				set_state_owner = ROOT
			}
		}
		s:STATE_SAVOY = {
			remove_homeland = cu:north_italian
		}
	}

	option = { # conduct a plebiscite
		name = natural_borders.3.b
		default_option = yes
		if = {
			limit = {
				is_owed_obligation_by = scope:savoy_country
			}
			scope:savoy_country = {
				set_owes_obligation_to = {
					country = ROOT
					setting = no
				}
			}
		}
		if = {
			limit = {
				has_global_variable = fra_savoy_sardinia
			}
			show_as_tooltip = {
				set_owes_obligation_to = {
					country = scope:savoy_country
					setting = yes
				}
			}
		}
		scope:savoy_state = {
			set_state_owner = ROOT
			set_state_type = unincorporated
			add_loyalists_in_state = {
				culture = cu:french
				value = very_large_radicals
			}
			add_loyalists_in_state = {
				culture = cu:francoprovencal
				value = very_large_radicals
			}
		}
		if = {
			limit = {
				exists = scope:nice_state
			}
			scope:nice_state = {
				set_state_owner = ROOT
			}
		}
		s:STATE_SAVOY = {
			remove_homeland = cu:north_italian
		}
	}
}
