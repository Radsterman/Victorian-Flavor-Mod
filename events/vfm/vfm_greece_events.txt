vfm_greece.7 = {
	type = country_event
	placement = scope:ottoman_state

	title = vfm_greece.7.t
	desc = vfm_greece.7.d
	flavor = vfm_greece.7.f

	gui_window = event_window_1char_tabloid
	left_icon = scope:historic_agitator_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	    exists = c:GRE
		this = c:GRE
		any_scope_character = {
			has_role = agitator
			has_ideology = ideology:ideology_ethno_nationalist
		}
	}

	immediate = {
		c:TUR = {
			save_scope_as = turkey_country
			random_scope_state = {
				limit = {
				    state_region = { 
					    is_homeland = cu:greek
					}
					OR = { 
					    has_state_trait = state_medeteranean_climate
						state_region = s:STATE_MACEDONIA
						state_region = s:STATE_WESTERN_THRACE
					}
				}
				save_scope_as = ottoman_state
			}
		}
		random_scope_character = {
			limit = {
				has_role = agitator
			    has_ideology = ideology:ideology_ethno_nationalist
			}
			save_scope_as = historic_agitator_scope
		}
	}

	option = {
		name = vfm_greece.7.a
		default_option = yes
		add_treasury = -30000
		scope:ottoman_state = {
		    add_radicals_in_state = {
				value = 0.15
				culture = cu:greek
			}
			state_region = {
			    add_devastation = 20
		    }
			
			trigger_event = { id = vfm_greece.8 }
		}
		show_as_tooltip = {
		    scope:ottoman_state = {
				if = { 
				    limit = { 
					    any_scope_building = {
						    is_building_type = building_barracks
					    }
					}
					remove_building = building_barracks
				}
			}
		}
		change_infamy = 5
	}
	
	option = {
		name = vfm_greece.7.b
		scope:ottoman_state = {
		    add_radicals_in_state = {
				value = 0.1
				culture = cu:greek
			}
			state_region = {
			       add_devastation = 10
		    }
			trigger_event = { id = vfm_greece.9 }
		}
		add_radicals = {
			pop_type = officers
			value = small_radicals
		}
	}
}
