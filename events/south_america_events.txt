# Peru-Bolivia success
south_america.1 = {
	type = country_event
	placement = ROOT
	title = south_america.1.t
	desc = south_america.1.d
	flavor =  south_america.1.f

	duration = 3

	event_image = {
		video = "southamerica_aristocrats"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
		# triggered by JE
	}

	immediate = {
		c:PEU = {
			save_scope_as = peru_country
			capital = {
				save_scope_as = peru_capital
			}
		}
	}

	option = { # Annex Peru
		name = south_america.1.a
		scope:peru_capital = {
			add_radicals_in_state = {
				value = 0.33
			}
		}
		if = {
			limit = {
				c:PEU = {
					is_player = no
				}
			}
			annex = c:PEU
		}
		ai_chance = {
			base = 1
		}
	}

	option = { #No
		name = south_america.1.b
		default_option = yes
		change_relations = {
			country = c:PEU
			value = 20
		}
		ai_chance = {
			base = 0
		}
	}
}

	}

		c:PEU = {
			save_scope_as = peru_country
		}
	}

	option = { #Yes
		name = south_america.2.a
		trigger = {
			c:PEU = {
				is_player = no
			}	
		}
		create_incident = {
			value = 50

		}
		create_diplomatic_pact = {
			country = c:PEU
			type = puppet
		}
		set_variable = {
			name = peru_bolivia_var
			value = 0
		}
		add_journal_entry = { type = je_peru_bolivia }
		ai_chance = {
			base = 1
		}
	}

	option = { #No
		name = south_america.2.b
		default_option = yes
		change_relations = {
			country = c:PEU
			value = 20
		}
		ai_chance = {
			base = 0
		}
	}
}

# Brazil starting war
south_america.3 = { #Ongoing Guerilla War 
	type = country_event
	placement = ROOT
	title = south_america.3.t
	desc = south_america.3.d
	flavor =  south_america.3.f

	duration = 3

	event_image = {
		video = "southamerica_war_civilians"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
		has_variable = recently_won_war
		any_state_region = {
			has_variable = brazil_sucession_state
		}
		country_is_in_south_america = yes
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = {
					has_variable = brazil_sucession_state
				}
			}
			save_scope_as = guerilla_state
		}
	}

	option = {
		name = south_america.3.a
		default_option = yes

		scope:guerilla_state.state_region = {
			add_devastation = 10
		}

		add_modifier = {
			name = modifier_destroy_guerillas
			months = normal_modifier_time
		}
	}

	option = {
		name = south_america.3.b

		scope:guerilla_state.state_region = {
			add_devastation = 50
		}

	}

}
