# Peru-Bolivia success
peru_bolivia_events.1 = {
	type = country_event
	placement = ROOT
	title = peru_bolivia_confederation.8.t
	desc = peru_bolivia_confederation.8.d
	flavor =  peru_bolivia_confederation.8.f

	duration = 3

	event_image = {
		video = "rp1_brazilian_city"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	trigger = {
		# triggered by JE
	}

	immediate = {
		if = {
			limit = {
				any_state = {
					state_region = s:STATE_AREQUIPA
					owner = {
						has_variable = pbc_member
					}
				}
			}
			random_state = {
				limit = {
					state_region = s:STATE_AREQUIPA
					owner = {
						has_variable = pbc_member
					}
				}
				save_scope_as = arequipa_state
			}
		}
		else = {
			ordered_state = {
				limit = {
					owner = {
						has_variable = pbc_member
					}
					state_is_in_south_america = yes
					is_incorporated = yes
				}
				save_scope_as = arequipa_state
			}
		}
	}

	option = { # Annex Peru
		name = peru_bolivia_confederation.8.a
		ai_chance = {
			base = 1
		}
		default_option = yes
		if = {
			limit = {
				c:NPU ?= {
					is_player = no
					is_subject_of = c:BOL
				}
			}
			annex = c:NPU
		}
		if = {
			limit = {
				c:SPU ?= {
					is_player = no
					is_subject_of = c:BOL
				}
			}
			annex = c:SPU
		}
		if = {
			limit = {
				c:IQU ?= {
					is_player = no
					is_subject_of = c:BOL
				}
			}
			annex = c:IQU
		}
		if = {
			limit = {
				c:PEU ?= {
					is_player = no
					is_subject_of = c:BOL
				}
			}
			annex = c:PEU
		}
		change_tag = PBC
		root = {
			save_scope_as = pbc_scope
		}
		every_country = {
			limit = {
				NOT = {
					THIS = ROOT
				}
				has_interest_marker_in_region = region_andes
			}
			post_notification = pbc_peru_bolivia_success
		}
	}
}

peru_bolivia_events.2 = {
	type = country_event
	placement = ROOT
	title = peru_bolivia_events.2.t
	desc = peru_bolivia_events.2.d
	flavor =  peru_bolivia_events.2.f

	duration = 3

	gui_window = event_window_1char_tabloid

	left_icon = scope:salaverry_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
		c:PEU ?= {
			is_subject_of = ROOT
		}
	}

	immediate = {
		c:PEU = {
			save_scope_as = peru_country
		}
		# Armed Forces
		create_character = {
			# Felipe Salaverry, was in the midst of a coup against Orbegoso
			template = PEU_Felipe_Salaverry
			on_created = {
				place_character_in_void = 6
			}
		}
		random_character_in_void = {
			limit = {
				has_template = PEU_Felipe_Salaverry
			}
			save_scope_as = salaverry_scope
		}
	}

	option = { #Yes
		name = peru_bolivia_events.2.a
		default_option = yes
		post_notification = bol_salaverry_death
		scope:salaverry_scope = {
			kill_character = yes
		}
		c:NPU = {
			add_radicals = {
				value = medium_radicals
			}
		}
		every_country = {
			limit = {
				country_is_in_south_america = yes
				has_diplomatic_relevance = root
				NOT = {
					has_variable = pbc_member
				}
				NOT = {
					is_country_type = decentralized
				}
			}
			trigger_event = { id = peru_bolivia_events.4 days = 7 popup = yes }
		}
		ai_chance = {
			base = 25
		}
	}

	option = { #No
		name = peru_bolivia_events.2.b
		hidden_effect = {
			scope:salaverry_scope = {
				remove_character_role = politician
				add_character_role = general
				free_character_from_void = yes
				exile_character_with_martyrdom = yes
			}
		}
		custom_tooltip = {
			text = salaverry_exile_tt
		}
		c:PEU = {
			add_radicals = {
				value = small_radicals
			}
		}
		every_country = {
			limit = {
				country_is_in_south_america = yes
				has_diplomatic_relevance = root
				NOT = {
					has_variable = pbc_member
				}
				NOT = {
					is_country_type = decentralized
				}
			}
			trigger_event = { id = peru_bolivia_events.4 days = 7 popup = yes }
		}
		ai_chance = {
			base = 5
		}
	}
}

			}
		}
		add_journal_entry = {
			type = je_war_of_the_confederation
		}
		add_modifier = {
			name = modifier_against_the_confederation
			months = short_modifier_time
		}
		if = { #Why? Because goal_add_value causes wonky things if we tie it directly to this variable. TODO revisit when support for a static target value in JEs.
			limit = {
				exists = global_var:peru_bolivia_var
			}
			set_variable = { name = peru_bolivia_opposition_var value = global_var:peru_bolivia_var }
		}
	}

	option = {
		name = peru_bolivia_events.4.b
		ai_chance = {
			base = 5
		}
		add_journal_entry = {
			type = je_war_of_the_confederation
		}
		change_relations = { country = c:BOL value = 15 }
		if = {
			limit = {
				exists = global_var:peru_bolivia_var
			}
			set_variable = { name = peru_bolivia_opposition_var value = global_var:peru_bolivia_var }
		}
	}
	
	option = {
		name = peru_bolivia_events.4.c
		trigger = {
			NOR = {
				c:CHL ?= this
				c:ARG ?= this
			}
		}
